Task 87: Implement Database Sharding

// Shard by userId % 3
const shards = [
  mysql.createPool({ host: 'shard1.db' }),
  mysql.createPool({ host: 'shard2.db' }),
  mysql.createPool({ host: 'shard3.db' })
];

function getShard(userId) {
  return shards[userId % shards.length];
}

// Insert to shard
app.post('/api/users', async (req, res) => {
  const userId = generateId();
  const shard = getShard(userId);
  await shard.execute('INSERT INTO Users SET?', { id: userId,...req.body });
  res.json({ userId });
});

// Read from shard
app.get('/api/users/:id', async (req, res) => {
  const shard = getShard(req.params.id);
  const [rows] = await shard.execute('SELECT * FROM Users WHERE id =?', [req.params.id]);
  res.json(rows[0]);
});
