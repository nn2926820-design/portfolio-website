Task 46: Implement Redis Caching

// Redis Client
const redis = require('redis');
const client = redis.createClient({ url: process.env.REDIS_URL });
client.connect();

// Cache Middleware
const cache = async (req, res, next) => {
  const key = `users:${req.params.id}`;
  const cached = await client.get(key);
  if (cached) {
    return res.json(JSON.parse(cached));
  }
  next();
};

// Use Cache
app.get('/api/users/:id', cache, async (req, res) => {
  const user = await User.findById(req.params.id);
  await client.setEx(`users:${req.params.id}`, 3600, JSON.stringify(user)); // 1hr
  res.json(user);
});

// Clear cache on update
app.put('/api/users/:id', async (req, res) => {
  const user = await User.findByIdAndUpdate(req.params.id, req.body, { new: true });
  await client.del(`users:${req.params.id}`);
  res.json(user);
});
