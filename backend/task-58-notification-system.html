Task 58: Build Notification System

// Notification Model + Socket.io
const Notification = mongoose.model('Notification', {
  userId: String,
  message: String,
  type: String,
  read: { type: Boolean, default: false },
  createdAt: { type: Date, default: Date.now }
});

// Send notification
const sendNotification = async (userId, message, type) => {
  const notif = await Notification.create({ userId, message, type });
  io.to(userId).emit('notification', notif);
};

// API Routes
app.get('/api/notifications', auth, async (req, res) => {
  const notifs = await Notification.find({ userId: req.user.id }).sort('-createdAt');
  res.json(notifs);
});

app.post('/api/notifications/read/:id', auth, async (req, res) => {
  await Notification.updateOne({ _id: req.params.id }, { read: true });
  res.json({ success: true });
});
