Task 65: Create Docker Setup

// Dockerfile
FROM node:18-alpine
WORKDIR /app
COPY package*.json./
RUN npm ci --only=production
COPY..
EXPOSE 5000
CMD ["node", "server.js"]

//.dockerignore
node_modules
.env
.git
*.log

// Build & Run
docker build -t myapp.
docker run -p 5000:5000 --env-file.env myapp
