Task 52: Create Background Job Queue

// Bull + Redis Queue
const Queue = require('bull');
const emailQueue = new Queue('email', { redis: { host: '127.0.0.1', port: 6379 } });

// Add job to queue
app.post('/api/send-bulk-email', async (req, res) => {
  const { users } = req.body;
  await emailQueue.add('send', { users }, {
    attempts: 3,
    backoff: 5000
  });
  res.json({ message: 'Jobs queued' });
});

// Process jobs
emailQueue.process('send', async (job) => {
  const { users } = job.data;
  for (const user of users) {
    await transporter.sendMail({ to: user.email, subject: 'Bulk Mail' });
  }
});
