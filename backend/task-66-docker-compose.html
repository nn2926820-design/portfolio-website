Task 66: Setup Docker Compose

// docker-compose.yml
version: '3.8'
services:
  app:
    build:.
    ports:
      - "5000:5000"
    env_file:.env
    depends_on:
      - db
      - redis
  db:
    image: mysql:8
    environment:
      MYSQL_ROOT_PASSWORD: secret
      MYSQL_DATABASE: myapp
    ports:
      - "3306:3306"
    volumes:
      - db_data:/var/lib/mysql
  redis:
    image: redis:alpine
    ports:
      - "6379:6379"
volumes:
  db_data:

// Run: docker-compose up -d
