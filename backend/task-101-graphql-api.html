// Apollo Server
const { ApolloServer, gql } = require('apollo-server-express');

const typeDefs = gql`
  type User {
    id: ID!
    name: String!
    email: String!
    posts: [Post]
  }
  type Post {
    id: ID!
    title: String!
    content: String
  }
  type Query {
    users: [User]
    user(id: ID!): User
  }
  type Mutation {
    createUser(name: String!, email: String!): User
  }
`;

const resolvers = {
  Query: {
    users: async () => await User.findAll(),
    user: async (_, { id }) => await User.findByPk(id)
  },
  Mutation: {
    createUser: async (_, { name, email }) => await User.create({ name, email })
  },
  User: {
    posts: async (user) => await Post.findAll({ where: { userId: user.id } })
  }
};

const server = new ApolloServer({ typeDefs, resolvers });
server.start().then(() => server.applyMiddleware({ app }));
