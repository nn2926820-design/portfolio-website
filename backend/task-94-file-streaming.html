Task 94: Implement File Streaming

// Stream large file
const fs = require('fs');

app.get('/api/download/:file', (req, res) => {
  const path = `./uploads/${req.params.file}`;
  const stat = fs.statSync(path);

  res.writeHead(200, {
    'Content-Type': 'application/octet-stream',
    'Content-Length': stat.size,
    'Content-Disposition': `attachment; filename=${req.params.file}`
  });

  const stream = fs.createReadStream(path);
  stream.pipe(res);
});

// Upload stream
app.post('/api/upload', (req, res) => {
  const stream = fs.createWriteStream(`./uploads/${Date.now()}`);
  req.pipe(stream);
  stream.on('finish', () => res.json({ success: true }));
});
