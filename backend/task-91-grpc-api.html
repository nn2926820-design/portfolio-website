Task 91: Implement gRPC API

// user.proto
syntax = "proto3";
service UserService {
  rpc GetUser (UserRequest) returns (UserReply) {}
}
message UserRequest { string id = 1; }
message UserReply { string name = 1; string email = 2; }

// server.js
const grpc = require('@grpc/grpc-js');
const protoLoader = require('@grpc/proto-loader');
const def = protoLoader.loadSync('user.proto');
const userProto = grpc.loadPackageDefinition(def).UserService;

function getUser(call, callback) {
  callback(null, { name: 'Naveen', email: 'test@mail.com' });
}

const server = new grpc.Server();
server.addService(userProto.service, { getUser });
server.bindAsync('0.0.0.0:50051', grpc.ServerCredentials.createInsecure(), () => {
  server.start();
});
