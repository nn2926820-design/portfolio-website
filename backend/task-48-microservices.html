Task 48: Create Microservices Architecture

// user-service - port 5001
app.get('/users/:id', async (req, res) => {
  const user = await User.findById(req.params.id);
  res.json(user);
});

// order-service - port 5002
app.get('/orders/:id', async (req, res) => {
  const order = await Order.findById(req.params.id);
  res.json(order);
});

// API Gateway - port 5000
const axios = require('axios');
app.get('/api/user-orders/:userId', async (req, res) => {
  const user = await axios.get(`http://localhost:5001/users/${req.params.userId}`);
  const orders = await axios.get(`http://localhost:5002/orders?userId=${req.params.userId}`);
  res.json({ user: user.data, orders: orders.data });
});
