Task 86: Setup Database Replication

// MySQL Master-Slave my.cnf
// Master config
[mysqld]
server-id = 1
log_bin = /var/log/mysql/mysql-bin.log
binlog_do_db = myapp

// Slave config
[mysqld]
server-id = 2
relay-log = /var/log/mysql/mysql-relay-bin.log
replicate-do-db = myapp

// Node.js read/write split
const mysql = require('mysql2');
const master = mysql.createPool({ host: 'master.db' });
const slave = mysql.createPool({ host: 'slave.db' });

const db = {
  write: (sql, params) => master.execute(sql, params),
  read: (sql, params) => slave.execute(sql, params)
};
