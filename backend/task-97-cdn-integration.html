Task 97: CDN Integration

// CloudFront + S3 URLs
const CDN_URL = process.env.CDN_URL || 'https://d123.cloudfront.net';

app.get('/api/assets/:file', (req, res) => {
  const cdnUrl = `${CDN_URL}/${req.params.file}`;
  res.redirect(cdnUrl);
});

// Cache-Control headers
app.use('/static', express.static('public', {
  maxAge: '30d',
  setHeaders: (res, path) => {
    res.setHeader('Cache-Control', 'public, max-age=2592000');
  }
}));

// Purge CDN cache
const AWS = require('aws-sdk');
const cloudfront = new AWS.CloudFront();
app.post('/api/purge-cdn', async (req, res) => {
  await cloudfront.createInvalidation({
    DistributionId: process.env.CF_DIST_ID,
    InvalidationBatch: {
      Paths: { Quantity: 1, Items: ['/*'] },
      CallerReference: Date.now().toString()
    }
  }).promise();
  res.json({ purged: true });
});
