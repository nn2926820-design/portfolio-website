Task 96: Image Optimization API

// Sharp image processing
const sharp = require('sharp');
const multer = require('multer');
const upload = multer({ storage: multer.memoryStorage() });

app.post('/api/upload-image', upload.single('image'), async (req, res) => {
  const optimized = await sharp(req.file.buffer)
  .resize(800, 600, { fit: 'inside' })
  .webp({ quality: 80 })
  .toBuffer();

  await fs.promises.writeFile(`./uploads/${Date.now()}.webp`, optimized);
  res.json({ success: true });
});

// Dynamic resize
app.get('/api/image/:file', async (req, res) => {
  const { w, h, q } = req.query;
  const img = await sharp(`./uploads/${req.params.file}`)
  .resize(parseInt(w), parseInt(h))
  .jpeg({ quality: parseInt(q) || 80 })
  .toBuffer();
  res.type('image/jpeg').send(img);
});
