Task 111: Push Notifications with FCM

// Firebase Admin for FCM
const admin = require('firebase-admin');

app.post('/api/send-notification', async (req, res) => {
  const { token, title, body } = req.body;
  const message = {
    notification: { title, body },
    token: token
  };
  await admin.messaging().send(message);
  res.json({ sent: true });
});

// Topic broadcast
app.post('/api/broadcast', async (req, res) => {
  await admin.messaging().send({
    notification: { title: req.body.title, body: req.body.body },
    topic: 'all_users'
  });
  res.json({ broadcast: true });
});

// Store FCM token
app.post('/api/register-token', auth, async (req, res) => {
  await User.updateOne(
    { _id: req.user.id },
    { $addToSet: { fcmTokens: req.body.token } }
  );
  res.json({ saved: true });
});
