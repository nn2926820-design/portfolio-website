// Sales aggregation
app.get('/api/stats/sales', async (req, res) => {
  const result = await client.search({
    index: 'orders',
    size: 0,
    aggs: {
      sales_per_month: {
        date_histogram: {
          field: 'createdAt',
          calendar_interval: 'month'
        },
        aggs: {
          total_sales: { sum: { field: 'amount' } },
          avg_order: { avg: { field: 'amount' } }
        }
      },
      top_products: {
        terms: { field: 'productId', size: 5 }
      }
    }
  });
  res.json(result.aggregations);
});
