Task 35: Develop Role-Based Access Control

// RBAC Middleware
const authorize = (...roles) => {
  return (req, res, next) => {
    if (!roles.includes(req.user.role)) {
      return res.status(403).json({ error: 'Forbidden' });
    }
    next();
  };
};

app.delete('/api/users/:id', auth, authorize('admin'), (req, res) => {
  res.json({ message: 'User deleted by admin' });
});

app.get('/api/reports', auth, authorize('admin', 'manager'), (req, res) => {
  res.json({ data: 'Reports' });
});
