Task 109: Email Template System

// Handlebars templates
const nodemailer = require('nodemailer');
const hbs = require('nodemailer-express-handlebars');

const transporter = nodemailer.createTransport({
  service: 'gmail',
  auth: { user: process.env.EMAIL, pass: process.env.EMAIL_PASS }
});

transporter.use('compile', hbs({
  viewEngine: { extname: '.hbs', layoutsDir: 'views/email/', defaultLayout: false },
  viewPath: 'views/email/',
  extName: '.hbs'
}));

// Send with template
app.post('/api/welcome', async (req, res) => {
  await transporter.sendMail({
    to: req.body.email,
    subject: 'Welcome!',
    template: 'welcome',
    context: { name: req.body.name, link: 'https://myapp.com' }
  });
  res.json({ sent: true });
});

// views/email/welcome.hbs
<h1>Hi {{name}}</h1>
<p>Welcome! <a href="{{link}}">Get Started</a></p>
